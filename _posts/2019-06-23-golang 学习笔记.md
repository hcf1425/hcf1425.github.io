## 8.pointer

```
	a := 5
	b := &a

	fmt.Println(a)

	fmt.Println(b) // memmory address of a 
	fmt.Printf("%T\n", b)

	// use * to read val from address 
	fmt.Println(*b)
	fmt.Println(*&a)

	// Change value with pointer
	*b = 10
	fmt.Println(*b)
	fmt.Println(*&a)
	fmt.Println(b)  
	// Didn't change the address

	// everything in go is pass by value !!!
}
```

```
package main
```

```
	ids := [] int {33, 76, 54, 23, 11, 2}
```

## 11.struct

```
package main 

import (
	"fmt"
	"strconv"
	)

//  Define person struct
type Person struct {
	firstName string
	lastName string
	city string
	gender string
	age int
}

// Greeting method (value recerver)
func (p Person) greet() string {
	return "Hello, my name is " + p.firstName + " " + p.lastName + "and my age is " +strconv.Itoa(p.age)
}

// hasBrithday method(pointer receiver)
func (p *Person)hasBrithday() {
	p.age++
}

// getMarried method (pointer receiver)
func (p *Person) getMarried( spouseLastName string) {
	if p.gender == "m" {
		return
	}else{
		p.lastName = spouseLastName
	}
}


func main() {
	// Init person using struct
	person1 := Person{firstName: "Samantha", lastName: "Smith", city: "Boston", gender: "f", age: 25}
	fmt.Println(person1)
	// Alternative
	person2 := Person{"Samantha", "Smith", "Boston", "f", 25}
	fmt.Println(person2)
	fmt.Println(person1.lastName)
	fmt.Println(person2.greet())
	person2.hasBrithday()
	fmt.Println(person2.age)  // 26
	person2.hasBrithday()
	person2.hasBrithday()  
	fmt.Println(person2.age) // 28

	// test getMarried method 
	person2.getMarried("Tomphson")
	fmt.Println(person2.greet())

}
```

```
package main 
import (
	"fmt"
	"math"
	)

type Shape interface {
	area() float64
}

type Circle struct {
	x, y, radius float64
}

type Rectangle struct {
	weidth, height float64
}

func (c Circle) area() float64 {
	return math.Pi * c.radius * c.radius
}

func (r Rectangle) area() float64 {
	return r.weidth * r.height
}

func getArea(s Shape) float64 {
	return s.area()
}

func main() { 
	circle := Circle{0, 0, 5}
	rectangle := Rectangle{10, 15}
	fmt.Printf("Circle Area: %f\n", getArea(circle))
	fmt.Printf("Rectangle Area: %f\n", getArea(rectangle))
}
```

## 14.channel
```
package main 
import "fmt"

func sum(s [] int , c chan int ){
	sum := 0
	for _, v := range(s){
		sum += v
	}
	c <- sum //put value sum into channel c
}

func main() {
	s := [] int {7, 2, 8, -9, 4, 0}
	c := make(chan int)
	fmt.Println(len(s))  // print the length of array s
	go sum(s[:len(s)/2], c)  // array slice 
	fmt.Println(s[:3])
	x := <-c
	go sum(s[len(s)/2:], c)
	y := <-c
	// x, y := <-c, <-c  // from channel c reveive value

	fmt.Println(x, y, x+y)
}
```

```
package main 

import "fmt"

// define a DivideError Struct
type DivideError struct {
	dividee int 
	divider int
}

// implement error interface 
func (de *DivideError) Error() string {
	strFormat := `
	Cannot proceed, the divider is zero.
	dividee: %d
	divider: 0
	`
	return fmt.Sprintf(strFormat, de.dividee)
}

// define a int type divide calculate function
func Divide(varDividee int, varDivider int) (result int, errorMsg string) {
	if varDivider == 0{
		dData := DivideError{
			dividee: varDividee,
			divider: varDivider,
		}
		errorMsg = dData.Error()
		return
	}else{
		return varDividee / varDivider, ""
	}
}

func main() {
	// normal case
	if result, errorMsg := Divide(100, 10); errorMsg == ""{
		fmt.Println("100/10 = ", result)
	}
	// unusual case
	if _, errorMsg := Divide(100, 0); errorMsg != ""  {
		fmt.Println("errorMsg is: ", errorMsg)
	}
}
```
